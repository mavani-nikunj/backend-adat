 // ✅ GET ALL TRANSACTIONS (WITH PAGINATION)
 [HttpGet]
 [Authorize]
 public async Task<IActionResult> GetAll(int? page, int? limit, string? paymenttype)
 {
     int clientId = User.GetClientId();
     int yearId = User.GetYearId();

     var query = _context.Transactions
         .Where(t => t.ClientId == clientId && t.YearId == yearId)
         .OrderByDescending(t => t.Id)
         .Select(t => new
         {
             t.Id,
             t.JvId,

             // ✅ PARTY
             Party = new
             {
                 Id = t.PartyId,
                 Name = _context.Partymasters
                     .Where(p => p.Id == t.PartyId)
                     .Select(p => p.Name)
                     .FirstOrDefault()
             },

             // ✅ CURRENCY
             Currency = new
             {
                 Id = t.CurrencyId,
                 Name = _context.Currencymsts
                     .Where(c => c.Id == t.CurrencyId)
                     .Select(c => c.Name)
                     .FirstOrDefault()
             },

             // ✅ CLIENT
             Client = new
             {
                 Id = t.ClientId,
                 Name = _context.Clientmasters
                     .Where(c => c.Id == t.ClientId)
                     .Select(c => c.Name)
                     .FirstOrDefault()
             },

             t.YearId,
             t.CreatDate,
             t.Amount,
             t.PaymentType,
             t.ExRate,
             t.ConvertAmount,
             t.Remark,
             t.Cdate,
             t.Udate,
             t.AdatAmount,
             t.AdatPercent,
             t.AmountWithoutAdat,
             t.BalanceType,
             t.InvoiceId
         })
         .AsQueryable();

     // ✅ APPLY TRANSACTION TYPE FILTER
     if (!string.IsNullOrWhiteSpace(paymenttype))
     {
         query = query.Where(t =>
             t.PaymentType.ToLower() == paymenttype.ToLower());
     }

     int totalRecords = await query.CountAsync();

     if (page.HasValue && limit.HasValue && page > 0 && limit > 0)
     {
         query = query
             .Skip((page.Value - 1) * limit.Value)
             .Take(limit.Value);
     }

     var data = await query.ToListAsync();

     if (!data.Any())
     {
         return Ok(new
         {
             Success = false,
             Message = "No transactions found"
         });
     }

     return Ok(new
     {
         Success = true,
         Message = "Transactions fetched successfully",
         TotalRecords = totalRecords,
         Page = page,
         Limit = limit,
         TotalPages = limit > 0
             ? (int)Math.Ceiling(totalRecords / (double)limit)
             : 1,
         Data = data
     });
